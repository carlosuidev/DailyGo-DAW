cerrarSesion()

Elimina las variables de sesión 
                
                    function cerrarSesion() {
                        try {
                            localStorage.clear();
                            window.location.href = "../../src/index.html";
                        } catch (error) {
                            console.log(
                                `Se ha producido un error a la hora de cerrar sesión: ${error}`
                            );
                        }
                    }
